struct TemplateView: View {
    // View constant that should be passed
    let subView: AnyView
    
    var body: some View {
        // Displays the view that has been passed
        subView()
    }
}

struct ContentView: View {
    var body: some View {
        // Calls the TemplateView and passes a subview
        TemplateView(subView: SomeSubview())
    }
}
