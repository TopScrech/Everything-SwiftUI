struct ContentView: View {
    @State private var showPopover = false
    
    var body: some View {
        Button("Open a sheet") {
            showPopover = true
        }
        .sheet(isPresented: $showPopover) {
            SheetView()
                .presentationDetents([.medium])
        }
    }
}

struct SheetView: View {
    @Environment(\.dismiss) private var dismiss
    
    var body: some View {
        Button("Dismiss the sheet") {
            dismiss()
        }
    }
}
