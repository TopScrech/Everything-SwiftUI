struct ContentView: View {
    @State private var showPopover = false
    
    var body: some View {
        Button("Open a popover") {
            showPopover = true
        }
        .popover(isPresented: $showPopover) {
            PopoverView()
        }
    }
}

struct PopoverView: View {
    @Environment(\.dismiss) private var dismiss
    
    var body: some View {
        Button("Dismiss the popover") {
            dismiss()
        }
    }
}
