struct ContentView: View {
    // Declaration of the colors used - Linear stops
    let gradientOne = Gradient(colors: [.green, .blue])
    
    // Declaration of the colors with custom stops
    let gradientOne = Gradient(stops: [
        .init(color: .green, location: 0.2),
        .init(color: .blue, location: 0.4)
    ])
    
    var body: some View {
        Rectangle()
            .fill(LinearGradient(gradient: gradientOne, startPoint: .top, endPoint: .bottom))
        
        Rectangle()
            .fill(AngularGradient(gradient: gradientOne, center: .center))
        
        Rectangle()
            .fill(RadialGradient(gradient: gradientOne, center: .center, startRadius: 0, endRadius: 100))
        
        Rectangle()
            .fill(EllipticalGradient(gradient: gradientOne, center: .top))
    }
}
